php日志监控

项目中没有对php日志进行监控，导致php fatal等处理不及时，自己写一个shell脚本，通过定时任务，对php日志进行监控，并发邮件给相应人员。

#!/bin/bash

#php日志目录
dir='php/'
log_f=$dir"php.log"
line_f=$dir"line.log"
#我们的php日志是按照天进行切割，所以需要一个记录天的文件
day_f=$dir"day.log"

today=`date +%Y%m%d`
if [ -f "$day_f" ];then
    lastday=`cat $day_f`
fi

total_l=`awk 'END{print NR}' $log_f`
if [ -f "$line_f" ];then
    last_l=`cat $line_f`
fi

if [[ "$today" -eq "$lastday" ]]; then
    read_l=$((total_l-last_l))
else    
    read_l=$total_l
fi

if [[ $read_l -gt 0 ]]; then
    echo "$total_l" >"$line_f"
    echo "$today" >"$day_f"
    content=`tail -n $read_l $log_f |grep -E "Warning|Fatal|error|PHP|#|Stack|trace|Exception|session|main|object|static|property|class|method|function|mysql|redis|mongo"`
    echo $content | mail -s 'PHP 日志监控' guanhua2011@gmail.com
fi
